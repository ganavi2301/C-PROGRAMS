// Program to print Hello World by using header file inclusion
#include <stdio.h>
int main()
{
	printf("Hello World!\n");
	return 0;
} 
o/p: Hello World!

// Program to print the value of PI using object like macro 
#include <stdio.h>
#define PI 3.14 //Object like macro
int main ()
{
	printf("The value of PI is %f\n", PI);
	return 0;
}
o/p: The value of PI is 3.140000

// Program to print square of a number using function like macro 
#include <stdio.h>
#define SQUARE(x) ((x)*(x)) //Function like macro
int main()
{	
	int num=4;
	printf("Square of %d is %d\n",num,SQUARE(num));
	return 0;
}
o/p: Square of 4 is 16

//Program using ifdef conditional compilation
#include <stdio.h>
#define SHOW
int main () 
{
#ifdef SHOW
	printf("This message is conditionally compiled!\n");
#endif
	return 0;
}
o/p: This message is conditionally compiled!
