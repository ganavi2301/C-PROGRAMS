//Program using calloc() function
#include <stdio.h>
#include<stdlib.h>
int main()
{
	int n,i,*ptr;
	printf("Enter total no of values:");
	scanf("%d",&n);
	ptr=(int*)calloc(n,size of int));
	printf("\n the entered values are:");
	for(i=0;i<n;i++)
	{
		printf("%d\t",*(ptr+i));
	}
}
