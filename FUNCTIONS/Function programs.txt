//Program to print sum of 2 numbers
#include<stdio.h>
void main()
{
	sum();
}
void sum()
{
	int a,b,sum=0;
	printf("Enter two numbers:");
	scanf("%d%d",&a,&b);
	sum=a+b;
	printf("sum=%d\n",sum)
}

//Program to implement call by value
#include<stdio.h>
void fun(int,int);
void main()
	{
		int x=5,y=7;
		fun(x,y);
		printf("inside main (calling function)\n");
		printf("x=%d y=%d\n",x,y);
	}
void fun(int x,int y)
	{
		int x=7;
		int y=5;
		printf("inside fun (called function)\n");
		printf("x=%d y=%d\n",x,y);
	}

//Program to implement call by reference
#include <stdio.h>
void fun(int*, int*);
void main()
{
	int x = 5, y = 7;
    	fun(&x, &y);
    	printf("Inside main (calling function)\n");
    	printf("x = %d  y = %d\n", x, y);
}

void fun(int *x, int *y)
{
    	*x = 7;
    	*y = 5;
    	printf("Inside fun (called function)\n");
    	printf("x = %d  y = %d\n", *x, *y);
}
