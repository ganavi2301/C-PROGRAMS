//Program using recursion
#include<stdio.h>
int sum(int x)
{
	int s=0;
	if(x==1)
	return x;
	else
	{
		s=x+sum(x-1);
		return s;
	}
}
void main()
{ 
	int a;
	a=sum(5);
	printf("%d",a);
}

//Program to print factorial of a number
#include <stdio.h>
int fun1(int);   
int fun2(int);   
void main()
{
    	printf("%d", fun1(5));  
}
int fun1(int n)
{
       	if (n <= 1)
       	return 1;
    	else
        return n*fun2(n-1);   
}
int fun2(int n)
{       
	if (n <= 1)
        return 1;
    	else
        return n * fun1(n - 1);   
}
